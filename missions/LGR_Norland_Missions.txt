			manpower_percentage = 0.3
		}

		effect = {
			add_prestige = 25
		}
	}

	lgr_norland_survive = {
		icon = mission_great_wall
		position = 3
		required_missions = {
			lgr_norland_end_disaster
		}

		trigger = {
			is_year = 1454
			owns_core_province = 35
			is_subject = no
		}

		effect = {
			curonia_area = {
				if = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}

			estonia_ingria_area = {
				if = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}

			livonia_area = {
				if = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}

			add_country_modifier = {
				name = lgr_norse_attack_livonia
				duration = 5475
			}
